1. Minikube

1.1 確保 Minikube 正常執行
    [安裝]
    winget source update
    winget install Kubernetes.minikube
    minikube start --driver=docker
        ⮕ 每次 docker_engine 啟動時要手動指令 minikube start

    [啟動]
    minikube status
    minikube update-context   #更新本地設定
    minikube start
        ⮕ 啟動正常時，本地 docker ps 可以看到 minikube 容器
        ⮕ 啟動正常時，kubectl get nodes 可以看到 minikube

    [重啟]
    minikube delete --all --purge   #刪除叢集
    minikube start --driver=docker

    [環境指派]
    kubectl config get-contexts   #查看叢集
    kubectl config use-context minikube   #切換叢集
        ⮕ 未來可能會切換為 GKE
    eval $(minikube docker-env)   #進入 minikube docker
        ⮕ 進入後的 docker ps 將看到許多個 k8s 容器
    eval $(minikube docker-env -u)   #離開 minikube docker
        ⮕ 離開至本地 docker ps 將看到一個 minikube 容器

    [運行服務]
    kubectl apply -f k8s/   #若有更新重新執行此
    kubectl get pods
    minikube service server-service   #啟動指定名稱服務
    kubectl delete -f k8s/   #徹底刪除所有設定及服務


1.2. 編譯映像
    [指令1: 至 minikube VM docker 內編譯]
    eval $(minikube docker-env)
    docker build -t tvs-server --target server .
    docker build -t tvs-scheduler --target scheduler .
    eval $(minikube docker-env -u)
    minikube image ls

    [指令2: 本地編譯後丟至 minikube]
    docker build -t tvs-server --target server .
    docker build -t tvs-scheduler --target scheduler .
    minikube image load tvs-server:latest
    minikube image load tvs-scheduler:latest
    minikube image ls


1.3 運行服務
    [指令: 運行與啟動服務]
    kubectl apply -f k8s/
    kubectl get pods
        ⮕ 如果服務有先後相依性，過早啟動的暫時 Error 是正常，等自動重啟即可
    minikube service server-service
        ⮕ 將自動依據服務 server-service 開啟映射


1.4 管理指令
    [指令: 資料庫管理]
    kubectl get pods -l app=mysql
        ⮕ 先取得 pod 名
    kubectl exec -i mysql-c76fd468f-d4mhr -- mysql -u root -ptt123456 twvixstock_db < scripts/twvixstock_db.sql
        ⮕ 運行 .sql
    kubectl exec -it mysql-c76fd468f-d4mhr -- mysql -u root -ptt123456 twvixstock_db -e "SELECT * FROM stock LIMIT 5;"
        ⮕ 執行 query

    [指令: 排程管理]
    kubectl get cronjobs
        ⮕ 所有將會開啟 jobs 的預期計畫
    kubectl get jobs
        ⮕ 查看所有 job 名，有名稱就可以查看 log
    kubectl create job --from=cronjob/tvs-scheduler-cron manualtest1
        ⮕ 立刻創建並只運行一次排程
    kubectl logs -l job-name=manualtest1
        ⮕ 查看log
    kubectl delete job manualtest1

    [指令: 服務管理]
    kubectl logs -l app=tvs-server
        ⮕ 查看指定服務所有 pod 運行 log
    kubectl logs -f -l app=tvs-server
        ⮕ 持續追蹤服務所有 pod 運行 log
    kubectl get pods -l app=tvs-server
        ⮕ 查看服務 pod 名
    kubectl logs tvs-server-5b4d99bbcd-s2xdw
        ⮕ 查看指定服務單一 pod 運行 log



2. GKE

2.1 專案創建
    [網站: 新建專案]
    https://console.cloud.google.com/
        ⮕ 新建專案 twvixstock

    [指令: 安裝 gcloud 與初始登入(一次性)]
    curl https://sdk.cloud.google.com | bash
        ⮕ windows要再建安裝資料路徑環境變數，安裝完成後cli要重啟
    gcloud init
    gcloud auth login

    [指令: 查看與切換專案]
    gcloud config get-value project
    gcloud config set project twvixstock
        ⮕ 切換專案有標籤建議不用理它


2.2 啟用API服務
    [指令: 啟動服務 Cloud Run Admin API + Cloud Build API+  Artifact Registry API + Kubernetes Engine API ]
    gcloud services enable run.googleapis.com
    gcloud services enable cloudbuild.googleapis.com
    gcloud services enable artifactregistry.googleapis.com
    gcloud services enable container.googleapis.com
    gcloud services list --enabled
        ⮕ 查看服務清單


2.3 建立 Artifact Registry
    [指令: 建立倉庫 tvs-repo]
    gcloud artifacts repositories create tvs-repo \
      --repository-format=docker \
      --location=asia-east1 \
      --description="docker repository"

    [指令: 設定 Docker 認證(一次性)]
    gcloud auth configure-docker asia-east1-docker.pkg.dev
        ⮕ 將自動修改設定  C:\Users\sdfsd\.docker\config.json  |  /home/sdfsd/.docker/config.json 
        ⮕ .docker\config.json 設定檔將增加 { "credHelpers": { "asia-east1-docker.pkg.dev": "gcloud" }


2.4 建立叢集
    [指令: 建立叢集 twvixstock-cluster ]
    gcloud container clusters create twvixstock-cluster \
        --region asia-east1-b \
        --num-nodes 2 \
        --disk-size 30
            ⮕ 地區不能只選 asia-east1 ，會創造 asia-east1-a/b/c 三個區
            ⮕ 預設100G，這邊使用30G (免費帳總空間上限為250G)

    [指令: 刪除叢集]
    gcloud container clusters delete twvixstock-cluster --zone asia-east1-b
        ⮕ 將刪除全部服務，就不會收費了

    [指令: 查看磁碟用量]
    gcloud config set project 別的專案
    gcloud compute disks list
        ⮕ 全部叢集磁碟用量，注意免費帳上限為 250G


2.5 部屬前確認
    [指令: 部屬環境確認]
    gcloud config get-value project
    gcloud config set project twvixstock
        ⮕ 確認並設定正確 GCP 專案
    kubectl config get-contexts
    kubectl config use-context gke開頭的
        ⮕ 確認並設定正確叢集
    docker ps   #是否在 minikube 內
    eval $(minikube docker-env -u)
        ⮕ 離開 minikube 環境


2.6 建置與部屬
    [指令: 掛載sql準備]
    kubectl delete configmap twvixstock-db-sql
    kubectl create configmap twvixstock-db-sql --from-file=scripts/twvixstock_db.sql

    [指令: 編譯與推送映像]
    docker build --no-cache -t asia-east1-docker.pkg.dev/twvixstock/tvs-repo/tvs-server:v1 --target server .
    docker build --no-cache -t asia-east1-docker.pkg.dev/twvixstock/tvs-repo/tvs-scheduler:v1 --target scheduler .
    docker push asia-east1-docker.pkg.dev/twvixstock/tvs-repo/tvs-server:v1
    docker push asia-east1-docker.pkg.dev/twvixstock/tvs-repo/tvs-scheduler:v1

    [指令: 運行與測試服務]
    kubectl apply -f k8s/
    kubectl get pods
    kubectl get services
        ⮕ 查看services EXTERNAL-IP(對外IP)
    kubectl create job --from=cronjob/tvs-scheduler-cron manualtest1
        ⮕ 立刻創建並只運行一次排程
    kubectl logs -l job-name=manualtest1
    kubectl delete job manualtest1

    [指令: 終止服務]
    kubectl delete -f k8s/
        ⮕ 終止所有部屬
    gcloud container clusters delete twvixstock-cluster --zone asia-east1-b
        ⮕ 刪除叢集

    [指令: 提交線上編譯]
    gcloud builds submit --config cloudbuild.yaml . \
        --substitutions=_TAG=v2


2.7 除錯與管理指令
    [指令: 當 pods 出錯時]
    kubectl get pods
    kubectl describe pods tvs-server-55869b9b79-8hpc6

    [指令: 當 image 出錯時]
    kubectl get deployment tvs-server -o yaml | grep image:
    kubectl get deployment tvs-scheduler -o yaml | grep image:

    [指令: 擴展 pods]
    kubectl scale deployment tvs-server --replicas=3
